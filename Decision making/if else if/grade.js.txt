var avg=Number(prompt("Enter average mark of student"));
if (avg < 60){
    alert("Grade : F");      
} 
else if (avg < 70){
     alert("Grade : D"); 
} 
else if (avg < 80){
      alert("Grade : C"); 
} 
else if (avg < 90) {
       alert("Grade : B"); 
} 
else if (avg < 100) {
       alert("Grade : A"); 
}
